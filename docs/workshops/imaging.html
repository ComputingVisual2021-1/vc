linkKernels

Se utilizaron kernels para el procesamiento de una imagen en dos formas, el filtro blur(volviendo la imagen borrosa) y el filtro sharp (acentuando los bordes de la imagen).

linkBlur kernel

1linklet v = 1.0 / 9.0;

2linklet kernel = [[ v, v, v ], [ v, v, v ], [ v, v, v ]]; 

3linkfunction preload() {

4link  img = loadImage("/vc/docs/sketches/medieval.jpg"); 

5link}

6link

7linkfunction setup() {

8link  createCanvas(1024, 512);

9link  noLoop();

10link}

11link

12linkfunction draw() {

13link  blurImg = createImage(img.width, img.height);

14link  blurImg.loadPixels();

15link  for (let x = 1; x < img.width; x++) {

16link    for (let y = 1; y < img.height; y++) {

17link      let sum = 0; 

18link      // kx, ky variables for iterating over the kernel

19link      // kx, ky have three different values: -1, 0, 1

20link      for (let kx = -1; kx <= 1; kx++) {

21link        for (let ky = -1; ky <= 1; ky++) {

22link          let xpos = x + kx;

23link          let ypos = y + ky;

24link          let val = red(img.get(xpos, ypos));

25link          // accumulate the  kernel sum

26link          // kernel is a 3x3 matrix

27link          // kx and ky have values -1, 0, 1

28link          // if we add 1 to kx and ky, we get 0, 1, 2

29link          // with that we can use it to iterate over kernel

30link          // and calculate the accumulated sum

31link          sum += kernel[kx+1][ky+1] * val;

32link        }

33link      }

34link      blurImg.set(x, y, color(sum));

35link    }

36link  }

37link  blurImg.updatePixels();

38link  image(img,0,0);

39link  image(blurImg, img.width, 0);

40link}
